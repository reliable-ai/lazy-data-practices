new_articles <- new_articles_raw %>% 
  mutate(
    dataset_name = if_else(
      is.na(our),
      paste(datasets, "(new)"),
      our
    )
  ) %>% 
  # Join in dataset IDs
  left_join(
    dataset_info %>% 
      # Keep only one row per name
      distinct(dataset_name, .keep_all = TRUE) %>% 
      select(dataset_name, dataset_id) %>%
      filter(!is.na(dataset_name)),
    by = c("our" = "dataset_name")
  )
